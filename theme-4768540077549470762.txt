// ---------- COUNTDOWN ----------
function getCountdown(ms){
let s=Math.floor(ms/1000);
let d=Math.floor(s/86400);s%=86400;
let h=Math.floor(s/3600);s%=3600;
let m=Math.floor(s/60);
let sec=s%60;
return `${d}d ${h}h ${m}m ${sec}s`;
}

📅 ${formatDisplay(r.date)} ⏰ ${r.time}<br>
⏳ <span class="timeLeft">${left}</span><br>

<div class="actionRow">
<button class="btn-done" onclick="complete(${r.id})">✔</button>
<button class="btn-edit" onclick="edit(${r.id})">✏</button>
<button class="btn-delete" onclick="del(${r.id})">❌</button>
<button class="btn-snooze" onclick="snooze(${r.id},5)">+5</button>
<button class="btn-snooze" onclick="snooze(${r.id},10)">+10</button>
</div>
</div>`;
});

calendar();
progressBox.innerText="Progress: "+getProgress()+"%";
}

// ---------- COUNTDOWN UPDATE ----------
function updateCountdownOnly(){
let now=Date.now();

document.querySelectorAll(".card").forEach(card=>{
let id=card.getAttribute("data-id");
let r=data.find(x=>x.id==id);
if(!r) return;

let left=getCountdown(Math.max(0,r.next-now));
let el = card.querySelector(".timeLeft");

if(el) el.innerText = left;
});
}

// ---------- SNOOZE ----------
function snooze(id,min=5){
let r=data.find(x=>x.id===id);

let d=new Date();
d.setMinutes(d.getMinutes()+min);

r.next=d.getTime();
r.date=format(d);
r.time=d.toTimeString().slice(0,5);

save();
}

// ---------- CALENDAR ----------
function calendar(){
let y=current.getFullYear();
let m=current.getMonth();
  
calendarHeader.innerHTML = `
<div class="calNav">
  <button class="prevBtn" onclick="prev()">Prev</button>
  <div class="monthText">${new Date(y,m).toLocaleString('default',{month:'long'})} ${y}</div>
  <button class="nextBtn" onclick="next()">Next</button>
</div>
`;

calendarGrid.innerHTML="";

let names=["Sun","Mon","Tue","Wed","Thu","Fri","Sat"];
names.forEach(d=>{
calendarGrid.innerHTML+=`<div style="text-align:center;font-size:12px">${d}</div>`;
});

let first=new Date(y,m,1).getDay();
let days=new Date(y,m+1,0).getDate();

for(let i=0;i<first;i++) calendarGrid.innerHTML+="<div></div>";

for(let d=1;d<=days;d++){
let dateStr=format(new Date(y,m,d));
let hasEvent=data.some(x=>x.date===dateStr && !x.completed);

calendarGrid.innerHTML+=`
<div class="day ${hasEvent?'hasEvent':''} ${selectedDate===dateStr?'selected':''}"
onclick="selectDate('${dateStr}')">${d}</div>`;
}
}

function next(){current.setMonth(current.getMonth()+1);render();}
